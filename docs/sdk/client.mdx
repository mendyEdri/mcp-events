---
title: Client SDK
description: Connect an agent and subscribe to events.
---

The repository includes a client package in `packages/client`.

## Example

```ts
import { ASPClient, WebSocketTransport } from '@esmcp/client';

const transport = new WebSocketTransport({
  url: 'ws://localhost:8080',
  reconnect: true
});

const client = new ASPClient({
  transport,
  clientInfo: { name: 'my-agent', version: '1.0.0' }
});

await client.connect();

const subscription = await client.subscribe({
  filter: {
    sources: ['github'],
    eventTypes: ['github.push']
  },
  delivery: {
    channels: ['websocket']
  }
});

client.onEvent('github.*', (event, subscriptionId) => {
  console.log(subscriptionId, event.type, event.data);
});

console.log('Subscription created:', subscription.id);
```
